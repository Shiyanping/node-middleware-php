 /**
 * Add控制器 添加图书控制器
 * @class
 */
class AddController {
  constructor() {

  }
  /**
   * 返回添加图书页面的操作
   */
  actionAdd() {
    return async (ctx, next) => {
      ctx.body = await ctx.render('add', {
        data: 'add Book'
      });
    }
  }
  /**
   * 测试数据
   * @return json
   */
  actionTest() {
    return async (ctx, next) => {
      ctx.body = {
        code: 0,
        message: '测试成功'
      }
    }
  }
}

module.exports = AddController;
